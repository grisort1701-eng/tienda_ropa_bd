SELECT c.id_cliente, c.nombre
FROM cliente c
WHERE NOT EXISTS (
  SELECT pr.id_producto
  FROM producto pr
  JOIN categoria cat
    ON pr.id_categoria = cat.id_categoria
  WHERE cat.nombre = 'Playeras'
  AND NOT EXISTS (
    SELECT 1
    FROM pedido p
    JOIN detalle_pedido d
      ON p.id_pedido = d.id_pedido
    WHERE p.id_cliente = c.id_cliente
      AND d.id_producto = pr.id_producto
  )
);
